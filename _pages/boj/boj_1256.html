문제의 풀이는 조합(Combination) 을 이용하여 풀었다.

가장 첫 자리부터 시작하여 첫 번째에 고르는 단어가 a이면

(a를 고를 수 있는 경우의 수 + z를 고를 수 있는 경우의 수a를 고를 수 있는 경우의 수 -1) \begin{pmatrix} \text{a를 고를 수 있는 경우의 수 + z를 고를 수 있는 경우의 수}\\ \text{a를 고를 수 있는 경우의 수 -1} \end{pmatrix} (a를 고를 수 있는 경우의 수 + z를 고를 수 있는 경우의 수a를 고를 수 있는 경우의 수 -1​)

이렇게 하는 것은 내가 지금 턴에 a a a 를 골라야 그 후에 무엇을 고르던 간에 k k k 에 맞는 경우를 찾을 수 있기 때문이다.

#include <iostream>
#include <string>
#define MAX 1000000000

using namespace std;

int n, m;
long long cache[400][400],k;
string target;

long long comb(int a, int b) {
    if (b == 0 || a == b||a<=1)return 1;
    if (cache[a][b] != 0)return cache[a][b];
    return cache[a][b] =comb(a - 1, b) + comb(a - 1, b - 1) < MAX? comb(a - 1, b) + comb(a - 1, b - 1) : MAX;
}

void f(long long idx, int a, int z) {
    if (a == 0) {
        while (z-- > 0)target+= "z";
        return;
    }
    if (z == 0) {
        while (a-- > 0)target += "a";
        return;
    }
    if (idx <= comb(a + z - 1, a - 1)) {
        target += "a";
        f(idx, a - 1, z);
    }
    else {
        target += "z";
        f(idx - comb(a + z - 1, a - 1), a, z - 1);
    }
}

int main() {
    cin >> n >> m >> k;
    if (comb(n+m, n) < k)cout << "-1";
    else {
        f(k, n, m);
        cout << target;
    }
    return 0;
}
