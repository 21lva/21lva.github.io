boj 1753

백준 1753번 문제

방향 그래프가 주어진 경우에 주어진 정점에서 다른 모든점으로의 최단경로를 구하는 문제이다.

간단하게 dijkstra algorithm 을 사용하면 된다.

#include <iostream>
#include <queue>
#include <vector>
#define INF 30000*20
using namespace std;


class edge {
public:
    int to;
    int value;
    edge(int t,int v):to(t),value(v){}
};
int key[20001];
struct comp {
    bool operator()(int a, int b) {
        return key[a] > key[b];
    }
};
int v, e,start;
vector<vector<edge> > edges;
priority_queue<int,vector<int>,comp> pq;

int main() {
    cin >> v >> e;
    cin >> start;
    for (int i = 0; i <= v; ++i) {
        edges.push_back(vector<edge>());
        key[i] = INF;
    }

    while (e-- > 0) {
        int from, to, weight;
        scanf("%d %d %d",&from,&to,&weight);
        edges.at(from).push_back(edge(to, weight));
    }
    key[start] = 0;
    pq.push(start);
    while (!pq.empty()) {
        int now = pq.top();
        pq.pop();
        for (edge adj : edges.at(now)) {
            if (key[adj.to] > key[now] + adj.value) {
                key[adj.to] = key[now] + adj.value;
                pq.push(adj.to);
            }
        }
    }
